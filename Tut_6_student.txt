\section{Dependence: Covariance, Correlation and the Hidden Link Between Risks}

\begin{MiniMissionBox}{Can two risky things become safer together?}
You will explore how two uncertain quantities move together, and why this matters for diversification.
\end{MiniMissionBox}

\begin{KeyResultBox}
For two random variables \(X\) and \(Y\),
\[
\operatorname{Cov}(X,Y) = E[(X-E[X])(Y-E[Y])] = E[XY]-E[X]E[Y].
\]
The correlation coefficient is
\[
\rho_{X,Y} = \frac{\operatorname{Cov}(X,Y)}{\sigma_X\sigma_Y}, \qquad -1\leq \rho_{X,Y}\leq 1.
\]
Covariance and correlation tell us whether two uncertain quantities tend to move together, move in opposite directions, or have little linear co-movement.
\end{KeyResultBox}

\begin{InvestigationBox}{Creating Two Artificial Assets With Chosen Correlation}{\stars{$\star\star\star$ Advanced}}
\textbf{Idea:} Instead of relying on real market data, we create artificial return data using the Uniform distribution. By carefully mixing a shared variable and an independent variable, we can control the correlation!

\medskip

\textbf{R Code:}
\begin{lstlisting}
set.seed(2026)
number_of_days <- 1000
chosen_correlation <- -0.5

# Step 1: Create a base sequence of uniform returns for Asset A.
# We center it at 0 so our correlation mixing math works cleanly.
asset_A_centered <- runif(
  n = number_of_days,
  min = -0.02,
  max = 0.02
)

# Step 2: Create independent noise using the same uniform spread.
independent_noise <- runif(
  n = number_of_days,
  min = -0.02,
  max = 0.02
)

# Step 3: Create Asset B by mixing Asset A and the noise.
# This mathematical trick guarantees the correlation we want!
asset_B_centered <- chosen_correlation * asset_A_centered +
  sqrt(1 - chosen_correlation^2) * independent_noise

# Step 4: Add a small average expected return to both (e.g., 0.05% daily)
asset_A_return <- asset_A_centered + 0.0005
asset_B_return <- asset_B_centered + 0.0005

# Check the generated correlation
cor(asset_A_return, asset_B_return)

plot(
  asset_A_return,
  asset_B_return,
  pch = 16,
  col = rgb(0, 0, 1, 0.35),
  main = "Simulated Asset Returns with Chosen Correlation",
  xlab = "Asset A Daily Return",
  ylab = "Asset B Daily Return"
)
abline(h = 0.0005, v = 0.0005, col = "gray")
\end{lstlisting}
\end{InvestigationBox}

\end{ReflectionBox}

\begin{CommonMistakeBox}
Correlation is not the same as causation. If two assets move together, it does not automatically mean one causes the other to move. There may be a shared economic factor, such as interest rates, inflation expectations or market sentiment.
\end{CommonMistakeBox}

\section{The Diversification Engine: Portfolio Variance}

\begin{MiniMissionBox}{Find the hidden engine of diversification}
You will test how changing correlation changes portfolio risk, even when the individual assets stay the same.
\end{MiniMissionBox}

\begin{KeyResultBox}
For a two-asset portfolio with weights \(w_A\) and \(w_B=1-w_A\), expected return is
\[
E[R_p] = w_AE[R_A]+w_BE[R_B].
\]
Portfolio variance is
\[
\sigma_p^2 = w_A^2\sigma_A^2 + w_B^2\sigma_B^2 + 2w_Aw_B\rho_{AB}\sigma_A\sigma_B.
\]
The final term,
\[
2w_Aw_B\rho_{AB}\sigma_A\sigma_B,
\]
is the \Blue{dependence term}. It is the mathematical reason why diversification may work or fail.
\end{KeyResultBox}

\begin{ExampleBox}{Portfolio Mean and Volatility Function}{\stars{$\star\star$ Intermediate}}
\textbf{R Code:}
\begin{lstlisting}
rm(list = ls())

calculate_portfolio_stats <- function(weight_asset1, weight_asset2, 
                                      return_asset1, return_asset2, 
                                      volatility_asset1, volatility_asset2, 
                                      correlation) {
  
  if (abs(weight_asset1 + weight_asset2 - 1) > 1e-6) {
    warning("Heads up: Your asset weights do not sum to 1 (100%).")
  }
  
  # Step 1: Expected Return
  expected_portfolio_return <- (weight_asset1 * return_asset1) + (weight_asset2 * return_asset2)
  
  # Step 2: Variance
  variance_asset1 <- volatility_asset1^2
  variance_asset2 <- volatility_asset2^2
  covariance <- volatility_asset1 * volatility_asset2 * correlation
  
  portfolio_variance <- (weight_asset1^2 * variance_asset1) + 
                        (weight_asset2^2 * variance_asset2) + 
                        (2 * weight_asset1 * weight_asset2 * covariance)
  
  # Step 3: Volatility
  portfolio_volatility <- sqrt(portfolio_variance)
  
  # Step 4: Output
  results <- list(
    Expected_Return = expected_portfolio_return,
    Volatility = portfolio_volatility
  )
  
  return(results)
}

my_portfolio <- calculate_portfolio_stats(
  weight_asset1 = 0.60, 
  weight_asset2 = 0.40,
  return_asset1 = 0.10, 
  return_asset2 = 0.05,
  volatility_asset1 = 0.15, 
  volatility_asset2 = 0.05,
  correlation = 0.20
)

print(my_portfolio)
\end{lstlisting}
\end{ExampleBox}

# install.packages("ggplot2")

# Load the library so we can use its plotting functions
library(ggplot2)

# 1. Define our asset characteristics (using decimals)
return_asset_A <- 0.12
volatility_asset_A <- 0.20
return_asset_B <- 0.08
volatility_asset_B <- 0.10

# 2. Define the correlations and the sequence of weights to test
correlations_to_test <- c(1, 0, -0.3, -1)

# seq() creates a sequence of numbers. Here, from 0 to 1 in steps of 0.01 (1%)
weights_asset_A <- seq(from = 0, to = 1, by = 0.01)

                     (2 * weight_A * weight_B * covariance)

      Volatility = port_volatility,
      Expected_Return = port_return,
      Correlation = as.character(cor) # Store as text so ggplot colors it cleanly
    )
    
    # Add this row to the bottom of our main table using rbind (Row Bind)

  }

\section{Minimum Variance Portfolio and Sharpe Ratio}

\begin{MiniMissionBox}{Choose the best portfolio, but define ``best'' first}
You will compare two ideas of optimality: lowest risk and best risk-adjusted return.
\end{MiniMissionBox}

\begin{KeyResultBox}
The \Blue{minimum variance portfolio} is the portfolio with the lowest volatility among the tested weights.

\medskip

The \Blue{Sharpe ratio} is
\[
\text{Sharpe Ratio} = \frac{E[R_p]-R_f}{\sigma_p}.
\]
It measures extra expected return per unit of volatility. A higher expected return is not automatically better if it requires too much risk.
\end{KeyResultBox}

\begin{itshape}
"Please extend the two-asset frontier code. For a chosen correlation, calculate the portfolio volatility, expected return and Sharpe ratio for every weight. Identify the minimum variance portfolio and the maximum Sharpe ratio portfolio. Use a 3\% risk-free rate."
\end{itshape}
\end{PromptBox}

\begin{InvestigationBox}{Comparing Two Kinds of Optimality}{\stars{$\star\star\star$ Advanced}}
\textbf{R Code:}
\begin{lstlisting}
# 1. Define our asset characteristics and the risk-free rate
return_asset_A <- 0.12
volatility_asset_A <- 0.20
return_asset_B <- 0.08
volatility_asset_B <- 0.10
risk_free_rate <- 0.03 # 3% Risk-Free Rate

# Let's choose a correlation of 0 (no relationship) for this analysis
chosen_correlation <- 0  

# 2. Create sequences for our weights (from 0 to 1 in steps of 0.01)
weights_asset_A <- seq(from = 0, to = 1, by = 0.01)

# Since weights must sum to 1, Asset B's weight is 1 minus Asset A's weight
# Notice we are subtracting an entire list of numbers from 1 all at once!
weights_asset_B <- 1 - weights_asset_A

# 3. Calculate metrics for ALL weight combinations simultaneously (Vectorization)

# Expected Return
portfolio_returns <- (weights_asset_A * return_asset_A) + (weights_asset_B * return_asset_B)

# Volatility
variance_A <- volatility_asset_A^2
variance_B <- volatility_asset_B^2
covariance <- volatility_asset_A * volatility_asset_B * chosen_correlation

portfolio_variances <- (weights_asset_A^2 * variance_A) + 
                       (weights_asset_B^2 * variance_B) + 
                       (2 * weights_asset_A * weights_asset_B * covariance)

portfolio_volatilities <- sqrt(portfolio_variances)

# 4. Calculate the Sharpe Ratio for every portfolio
# Formula: (Expected Return - Risk-Free Rate) / Volatility
sharpe_ratios <- (portfolio_returns - risk_free_rate) / portfolio_volatilities

# 5. Combine everything into a neat master table (Data Frame)
portfolio_data <- data.frame(
  Weight_Asset_A = weights_asset_A,
  Weight_Asset_B = weights_asset_B,
  Expected_Return = portfolio_returns,
  Volatility = portfolio_volatilities,
  Sharpe_Ratio = sharpe_ratios
)

# 6. Identify the Minimum Variance Portfolio (MVP)
# which.min() finds the exact row index where Volatility is the lowest
row_mvp <- which.min(portfolio_data$Volatility)
min_variance_portfolio <- portfolio_data[row_mvp, ] # Extract that specific row

# 7. Identify the Maximum Sharpe Ratio Portfolio (The Optimal Risky Portfolio)
# which.max() finds the exact row index where the Sharpe Ratio is the highest
row_max_sharpe <- which.max(portfolio_data$Sharpe_Ratio)
max_sharpe_portfolio <- portfolio_data[row_max_sharpe, ] # Extract that specific row

# 8. Print the final discoveries!
cat("\n--- Minimum Variance Portfolio (Lowest Risk) ---\n")
print(min_variance_portfolio)

cat("\n--- Maximum Sharpe Ratio Portfolio (Best Risk-Adjusted Return) ---\n")
print(max_sharpe_portfolio)
\end{lstlisting}
\end{InvestigationBox}

\section{Challenge Extension: When Diversification Fails}

\begin{MiniMissionBox}{Stress test the comfort zone}
You will check how the same portfolio can become riskier when correlation changes during stress.
\end{MiniMissionBox}

\begin{KeyResultBox}
Diversification works best when risks are not perfectly positively correlated. It can fail when many assets become exposed to the same hidden factor. In market crises, assets that seemed different may suddenly move together.
\end{KeyResultBox}

\begin{InvestigationBox}{Stress Testing Correlation}{\stars{$\star\star\star$ Challenge}}
\textbf{Task.} Suppose a portfolio was designed under correlation \(-0.3\), but during a stress period the correlation becomes \(+0.8\). Compare the volatility of the same portfolio weight under both assumptions.

\medskip

# 4. Calculate Variance and Volatility for the NORMAL scenario

\end{lstlisting}
\end{InvestigationBox}

\section{Real-World 4-Asset Portfolio \& Monte Carlo Simulation}

\begin{MiniMissionBox}{From classroom model to market experiment}
This optional extension uses real data and random weights to explore many possible 4-asset portfolios.
\end{MiniMissionBox}

In this session, we transition from theoretical 2-asset models to evaluating a 4-asset portfolio using real market data. We will calculate the portfolio's expected return and volatility without using complex matrix algebra, and then deploy a Monte Carlo simulation to uncover the Efficient Frontier.

\subsection{Task 1: Evaluating a 4-Asset Portfolio Using Real Data}

Using the \texttt{quantmod} package, we can pull historical stock data and calculate daily returns. By blending these individual returns into a single "Portfolio Daily Return" timeline, we completely bypass the need for a complex covariance matrix.

\begin{lstlisting}[language=R, caption=4-Asset Portfolio Evaluation without Matrices]
# Load the library to fetch financial data
library(quantmod)

# 1. Download real stock data (AAPL, MSFT, GOOG, AMZN)
getSymbols(c("AAPL", "MSFT", "GOOG", "AMZN"), from = "2023-01-01", to = "2024-01-01")

# 2. Extract Adjusted Close prices and calculate daily returns
# ROC stands for Rate of Change
returns_AAPL <- ROC(Ad(AAPL), type = "discrete", na.pad = FALSE)
returns_MSFT <- ROC(Ad(MSFT), type = "discrete", na.pad = FALSE)
returns_GOOG <- ROC(Ad(GOOG), type = "discrete", na.pad = FALSE)
returns_AMZN <- ROC(Ad(AMZN), type = "discrete", na.pad = FALSE)

# 3. Define equal weights (25% each, must sum to 1.0)
w_AAPL <- 0.25; w_MSFT <- 0.25; w_GOOG <- 0.25; w_AMZN <- 0.25

# 4. Calculate the DAILY portfolio return timeline
port_daily_returns <- (w_AAPL * returns_AAPL) + (w_MSFT * returns_MSFT) + 
                      (w_GOOG * returns_GOOG) + (w_AMZN * returns_AMZN)

# 5. Annualize Expected Return and Volatility (assuming 252 trading days)
annual_return <- mean(port_daily_returns) * 252
annual_vol <- sd(port_daily_returns) * sqrt(252)

# Print results
cat("Expected Annual Return:", round(annual_return * 100, 2), "%\n")
cat("Annualized Volatility (Risk):", round(annual_vol * 100, 2), "%\n")
\end{lstlisting}

\subsection{Task 2: Monte Carlo Simulation for the Efficient Frontier}

Calculating every exact weight combination for 4 assets is mathematically overwhelming. Instead, we use a Monte Carlo simulation to generate thousands of random weight combinations. The upper-left boundary of the resulting "cloud" of portfolios will naturally reveal the Efficient Frontier.

\begin{lstlisting}[language=R, caption=Monte Carlo Efficient Frontier Simulation]
library(ggplot2)

# Define simulation parameters
num_portfolios <- 2500
sim_returns <- numeric(num_portfolios)
sim_volatilities <- numeric(num_portfolios)
sim_sharpes <- numeric(num_portfolios)
risk_free_rate <- 0.03

# Run the Monte Carlo Simulation
for (i in 1:num_portfolios) {
  
  # Generate 4 random numbers and normalize them so they sum to 1 (100%)
  rands <- runif(4, min = 0, max = 1)
  weights <- rands / sum(rands)
  
  # Calculate daily portfolio returns using the timeline approach
  daily_ret <- (weights[1] * returns_AAPL) + (weights[2] * returns_MSFT) + 
               (weights[3] * returns_GOOG) + (weights[4] * returns_AMZN)
  
  # Annualize and store the risk/return metrics
  ann_ret <- mean(daily_ret, na.rm = TRUE) * 252
  ann_vol <- sd(daily_ret, na.rm = TRUE) * sqrt(252)
  
  sim_returns[i] <- ann_ret
  sim_volatilities[i] <- ann_vol
  sim_sharpes[i] <- (ann_ret - risk_free_rate) / ann_vol
}

# Combine simulated data into a Data Frame for Plotting
portfolio_cloud <- data.frame(Volatility = sim_volatilities, 
                              Expected_Return = sim_returns, 
                              Sharpe_Ratio = sim_sharpes)

# Plot the Efficient Frontier Cloud
ggplot(portfolio_cloud, aes(x = Volatility, y = Expected_Return, color = Sharpe_Ratio)) +
  geom_point(alpha = 0.5, size = 1.5) + 
  scale_color_viridis_c(option = "plasma") + 
  labs(title = "4-Asset Efficient Frontier (Monte Carlo Simulation)",
       subtitle = "Simulating 2,500 random tech portfolios",
       x = "Annualized Volatility (Risk)",
       y = "Annualized Expected Return",
       color = "Sharpe Ratio") +

\section{Consolidation Questions}

\begin{InvestigationBox}{Review Set A: Risk Register and Scoring}{\stars{$\star$ Core}}
\begin{enumerate}
\item Write two new risk statements using the structure ``If \dots, then \dots''.
\item Assign likelihood and impact scores from 1 to 3.
\item Explain why the risk score is useful but imperfect.
\item Identify one treatment for each risk.
\item Explain the difference between inherent risk and residual risk.
\end{enumerate}
\end{InvestigationBox}

\begin{InvestigationBox}{Review Set B: Frequency and Severity}{\stars{$\star\star$ Intermediate}}
\begin{enumerate}
\item Define frequency and severity in the claim simulation.
\item Explain why total loss is random.
\item Interpret the mean, standard deviation and 95th percentile of simulated total losses.
\item \textbf{Explain why a Uniform distribution might not be realistic for insurance severity (e.g., real claims have no strict upper limit and tend to be skewed). }
\item Describe one limitation of our Binomial frequency model.
\end{enumerate}
\end{InvestigationBox}

\begin{InvestigationBox}{Review Set C: Correlation and Portfolio Risk}{\stars{$\star\star\star$ Advanced}}
\begin{enumerate}
\item State the two-asset portfolio variance formula.
\item Identify the covariance or correlation term in the formula.
\item Explain why lower correlation can reduce portfolio volatility.
\item Compare the meaning of minimum variance and maximum Sharpe ratio.
\item Explain how stress testing correlation helps risk management.
\end{enumerate}
\end{InvestigationBox}

\begin{InvestigationBox}{Challenge Questions}{\stars{$\star\star\star$ Challenge}}
\begin{enumerate}
\item If two risks have correlation zero, are they necessarily independent? Explain.
\item Can diversification eliminate all risk in real financial markets? Why or why not?
\item Why might a low-return asset still appear in a good portfolio?
\item How could model error affect an efficient frontier?
\item How would transaction costs, liquidity risk or short-selling constraints change the decision?
\end{enumerate}
\end{InvestigationBox}
